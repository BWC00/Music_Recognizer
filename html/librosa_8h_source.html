96 static Matrixcf stft(Vectorf &x, int n_fft, int n_hop, const std::string &win, bool center, const std::string &mode){
97  // hanning
98  Vectorf window = 0.5*(1.f-(Vectorf::LinSpaced(n_fft, 0.f, static_cast<float>(n_fft-1))*2.f*M_PI/n_fft).array().cos());
99 
100  int pad_len = center ? n_fft / 2 : 0;
101  Vectorf x_paded = pad(x, pad_len, pad_len, mode, 0.f);
102 
103  int n_f = n_fft/2+1;
104  int n_frames = 1+(x_paded.size()-n_fft) / n_hop;
105  Matrixcf X(n_frames, n_fft);
106  Eigen::FFT<float> fft;
107 
108  for (int i = 0; i < n_frames; ++i){
109  Vectorf x_frame = window.array()*x_paded.segment(i*n_hop, n_fft).array();
110  X.row(i) = fft.fwd(x_frame);
111  }
112  return X.leftCols(n_f);
113 }

119 static Matrixf melfilter(int sr, int n_fft, int n_mels, int fmin, int fmax){
120  int n_f = n_fft/2+1;
121  Vectorf fft_freqs = (Vectorf::LinSpaced(n_f, 0.f, static_cast<float>(n_f-1))*sr)/n_fft;
122 
123  float f_min = 0.f;
124  float f_sp = 200.f/3.f;
125  float min_log_hz = 1000.f;
126  float min_log_mel = (min_log_hz-f_min)/f_sp;
127  float logstep = logf(6.4f)/27.f;
128 
129  auto hz_to_mel = [=](int hz, bool htk = false) -> float {
130  if (htk){
131  return 2595.0f*log10f(1.0f+hz/700.0f);
132  }
133  float mel = (hz-f_min)/f_sp;
134  if (hz >= min_log_hz){
135  mel = min_log_mel+logf(hz/min_log_hz)/logstep;
136  }
137  return mel;
138  };
139  auto mel_to_hz = [=](Vectorf &mels, bool htk = false) -> Vectorf {
140  if (htk){
141  return 700.0f*(Vectorf::Constant(n_mels+2, 10.f).array().pow(mels.array()/2595.0f)-1.0f);
142  }
143  return (mels.array()>min_log_mel).select(((mels.array()-min_log_mel)*logstep).exp()*min_log_hz, (mels*f_sp).array()+f_min);
144  };
145 
146  float min_mel = hz_to_mel(fmin);
147  float max_mel = hz_to_mel(fmax);
148  Vectorf mels = Vectorf::LinSpaced(n_mels+2, min_mel, max_mel);
149  Vectorf mel_f = mel_to_hz(mels);
150  Vectorf fdiff = mel_f.segment(1, mel_f.size() - 1) - mel_f.segment(0, mel_f.size() - 1);
151  Matrixf ramps = mel_f.replicate(n_f, 1).transpose().array() - fft_freqs.replicate(n_mels + 2, 1).array();
152 
153  Matrixf lower = -ramps.topRows(n_mels).array()/fdiff.segment(0, n_mels).transpose().replicate(1, n_f).array();
154  Matrixf upper = ramps.bottomRows(n_mels).array()/fdiff.segment(1, n_mels).transpose().replicate(1, n_f).array();
155  Matrixf weights = (lower.array()<upper.array()).select(lower, upper).cwiseMax(0);
156 
157  auto enorm = (2.0/(mel_f.segment(2, n_mels)-mel_f.segment(0, n_mels)).array()).transpose().replicate(1, n_f);
158  weights = weights.array()*enorm;
159 
160  return weights;
161 }

179 static Matrixf dct(Matrixf& x, bool norm, int type) {
180  int N = x.cols();
181  Matrixf xi = Matrixf::Zero(N, N);
182  xi.rowwise() += Vectorf::LinSpaced(N, 0.f, static_cast<float>(N-1));
183  // type 2
184  Matrixf coeff = 2*(M_PI*xi.transpose().array()/N*(xi.array()+0.5)).cos();
185  Matrixf dct = x*coeff.transpose();
186  // ortho
187  if (norm) {
188  Vectorf ortho = Vectorf::Constant(N, std::sqrt(0.5f/N));
189  ortho[0] = std::sqrt(0.25f/N);
190  dct = dct*ortho.asDiagonal();
191  }
192  return dct;
193 }
